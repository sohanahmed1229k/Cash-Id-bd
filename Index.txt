<script>
    // --- DATA INITIALIZATION & LOCAL STORAGE ---
    const DB_KEY = 'buyBotDB';
    let db = {
        users: [],
        items: [],
        purchaseRequests: [],
        withdrawalRequests: []
    };
    let currentUser = null;
    let nextId = 1;

    function initializeDB() {
        const storedDB = localStorage.getItem(DB_KEY);
        if (storedDB) {
            db = JSON.parse(storedDB);
        } else {
            // Default data for first time
            db.users = [{ id: nextId++, username: 'admin', password: 'admin123', balance: 0, isAdmin: true }];
            db.items = [
                { id: nextId++, name: 'অফার সম্পন্ন করুন - ১', price: 1000 },
                { id: nextId++, name: 'গেম খেলে লেভেল আপ করুন', price: 1500 },
                { id: nextId++, name: 'অ্যাপ ডাউনলোড করুন', price: 500 }
            ];
            saveDB();
        }
        const allIds = [
            ...db.users.map(u => u.id),
            ...db.items.map(i => i.id),
            ...db.purchaseRequests.map(r => r.id),
            ...db.withdrawalRequests.map(r => r.id)
        ];
        nextId = allIds.length > 0 ? Math.max(...allIds) + 1 : 1;
    }

    function saveDB() {
        localStorage.setItem(DB_KEY, JSON.stringify(db));
    }

    // --- DOM ELEMENTS ---
    const authPage = document.getElementById('authPage');
    const userDashboard = document.getElementById('userDashboard');
    const adminDashboard = document.getElementById('adminDashboard');
    const loginForm = document.getElementById('loginForm');
    const signupForm = document.getElementById('signupForm');
    const messageBox = document.getElementById('messageBox');

    // --- CORE FUNCTIONS ---
    function showMessage(text, type = 'info') {
        messageBox.innerHTML = `<div class="message ${type}">${text}</div>`;
        setTimeout(() => { messageBox.innerHTML = ''; }, 4000);
    }

    function showPage(page) {
        authPage.classList.add('hidden');
        userDashboard.classList.add('hidden');
        adminDashboard.classList.add('hidden');
        page.classList.remove('hidden');
    }

    function showAuthTab(tab) {
        const tabs = document.querySelectorAll('.auth-tab');
        tabs.forEach(t => t.classList.remove('active'));
        if (tab === 'login') {
            tabs[0].classList.add('active');
            loginForm.classList.remove('hidden');
            signupForm.classList.add('hidden');
        } else {
            tabs[1].classList.add('active');
            loginForm.classList.add('hidden');
            signupForm.classList.remove('hidden');
        }
    }

    function updateDashboard() {
        if (!currentUser) return;
        if (currentUser.isAdmin) {
            showPage(adminDashboard);
            displayAdminPurchaseRequests();
            displayAdminWithdrawalRequests();
            displayAllUsers();
            displayAllItemsForAdmin();
            populateUserSelect();
        } else {
            showPage(userDashboard);
            document.getElementById('userWelcome').textContent = `স্বাগতম, ${currentUser.username}`;
            document.getElementById('userBalanceDisplay').textContent = `ব্যালেন্স: ${currentUser.balance} BDT`;
            displayItems();
            displayUserPurchaseRequests();
        }
    }

    // --- AUTHENTICATION ---
    loginForm.addEventListener('submit', (e) => {
        e.preventDefault();
        const username = document.getElementById('loginUsername').value;
        const password = document.getElementById('loginPassword').value;
        const user = db.users.find(u => u.username === username && u.password === password);
        if (user) {
            currentUser = user;
            showMessage('সফলভাবে লগইন হয়েছে!', 'success');
            updateDashboard();
        } else {
            showMessage('ভুল ইউজারনেম বা পাসওয়ার্ড!', 'error');
        }
    });

    signupForm.addEventListener('submit', (e) => {
        e.preventDefault();
        const username = document.getElementById('signupUsername').value;
        const password = document.getElementById('signupPassword').value;
        if (db.users.some(u => u.username === username)) {
            showMessage('এই ইউজারনেমটি ইতিমধ্যেই ব্যবহৃত হচ্ছে!', 'error');
            return;
        }
        const newUser = { id: nextId++, username, password, balance: 0, isAdmin: false };
        db.users.push(newUser);
        saveDB();
        showMessage('অ্যাকাউন্ট সফলভাবে তৈরি হয়েছে! এখন লগইন করুন।', 'success');
        showAuthTab('login');
        signupForm.reset();
    });

    function logout() {
        currentUser = null;
        showPage(authPage);
        showMessage('আপনি লগআউট হয়েছেন।', 'info');
        loginForm.reset();
    }

    // --- USER FUNCTIONS ---
    function displayItems() {
        const itemList = document.getElementById('itemList');
        itemList.innerHTML = '';
        db.items.forEach(item => {
            itemList.innerHTML += `
                <div class="card">
                    <h3>${item.name}</h3>
                    <div class="reward">পুরস্কার: ${item.price} BDT</div>
                    <button onclick="buyItem(${item.id})">সম্পন্ন করুন</button>
                </div>
            `;
        });
    }

    function buyItem(itemId) {
        const item = db.items.find(i => i.id === itemId);
        const newRequest = {
            id: nextId++,
            user: currentUser.username,
            itemName: item.name,
            price: item.price, // This 'price' is the reward amount
            status: 'pending'
        };
        db.purchaseRequests.push(newRequest);
        saveDB();
        showMessage(`'${item.name}' সম্পন্ন করার জন্য রিকোয়েস্ট পাঠানো হয়েছে।`, 'info');
        updateDashboard();
    }

    function displayUserPurchaseRequests() {
        const userRequestsDiv = document.getElementById('userPurchaseRequests');
        const userRequests = db.purchaseRequests.filter(r => r.user === currentUser.username);
        if (userRequests.length === 0) {
            userRequestsDiv.innerHTML = '<p>আপনার কোনো পুরস্কারের রিকোয়েস্ট নেই।</p>';
            return;
        }
        let html = `<table class="data-table"><tr><th>অফার/টাস্ক</th><th>পুরস্কার</th><th>অবস্থা</th></tr>`;
        userRequests.forEach(req => {
            const statusText = req.status === 'pending' ? 'অপেক্ষমান' : (req.status === 'approved' ? 'অনুমোদিত' : 'বাতিল');
            html += `<tr><td>${req.itemName}</td><td>${req.price} BDT</td><td>${statusText}</td></tr>`;
        });
        html += `</table>`;
        userRequestsDiv.innerHTML = html;
    }

    function requestWithdrawal() {
        const amount = parseInt(document.getElementById('withdrawAmount').value);
        if (!amount || amount <= 0) { showMessage('সঠিক পরিমাণ লিখুন।', 'error'); return; }
        if (amount > currentUser.balance) { showMessage('আপনার অপর্যাপ্ত ব্যালেন্স!', 'error'); return; }

        const newRequest = { id: nextId++, user: currentUser.username, amount, status: 'pending' };
        db.withdrawalRequests.push(newRequest);
        saveDB();
        showMessage(`${amount} BDT উইথড্রের জন্য রিকোয়েস্ট পাঠানো হয়েছে।`, 'info');
        document.getElementById('withdrawAmount').value = '';
        updateDashboard();
    }

    // --- ADMIN FUNCTIONS ---
    function displayAdminPurchaseRequests() {
        const adminRequestsDiv = document.getElementById('adminPurchaseRequests');
        const pendingRequests = db.purchaseRequests.filter(r => r.status === 'pending');
        if (pendingRequests.length === 0) { adminRequestsDiv.innerHTML = '<p>কোনো অপেক্ষমান রিকোয়েস্ট নেই।</p>'; return; }
        let html = `<table class="data-table"><tr><th>ইউজার</th><th>অফার/টাস্ক</th><th>পুরস্কার</th><th>অ্যাকশন</th></tr>`;
        pendingRequests.forEach(req => {
            html += `<tr><td>${req.user}</td><td>${req.itemName}</td><td>${req.price} BDT</td><td class="actions">
                <button class="approve" onclick="approvePurchase(${req.id})">অনুমোদন</button>
                <button class="reject" onclick="rejectPurchase(${req.id})">বাতিল</button></td></tr>`;
        });
        html += `</table>`;
        adminRequestsDiv.innerHTML = html;
    }

    function approvePurchase(reqId) {
        const request = db.purchaseRequests.find(r => r.id === reqId);
        if (request) {
            const user = db.users.find(u => u.username === request.user);
            if (user) {
                user.balance += request.price; // *** KEY CHANGE: ADD REWARD TO BALANCE ***
                request.status = 'approved';
                saveDB();
                showMessage(`${request.user}-কে ${request.price} BDT পুরস্কার দেওয়া হয়েছে।`, 'success');
                updateDashboard();
            }
        }
    }
    function rejectPurchase(reqId) {
        const request = db.purchaseRequests.find(r => r.id === reqId);
        if (request) { request.status = 'rejected'; saveDB(); showMessage(`রিকোয়েস্ট বাতিল।`, 'error'); updateDashboard(); }
    }

    function displayAdminWithdrawalRequests() {
        const adminRequestsDiv = document.getElementById('adminWithdrawalRequests');
        const pendingRequests = db.withdrawalRequests.filter(r => r.status === 'pending');
        if (pendingRequests.length === 0) { adminRequestsDiv.innerHTML = '<p>কোনো অপেক্ষমান উইথড্র রিকোয়েস্ট নেই।</p>'; return; }
        let html = `<table class="data-table"><tr><th>ইউজার</th><th>পরিমাণ</th><th>অ্যাকশন</th></tr>`;
        pendingRequests.forEach(req => {
            html += `<tr><td>${req.user}</td><td>${req.amount} BDT</td><td class="actions">
                <button class="approve" onclick="approveWithdrawal(${req.id})">অনুমোদন</button>
                <button class="reject" onclick="rejectWithdrawal(${req.id})">বাতিল</button></td></tr>`;
        });
        html += `</table>`;
        adminRequestsDiv.innerHTML = html;
    }

    function approveWithdrawal(reqId) {
        const request = db.withdrawalRequests.find(r => r.id === reqId);
        if (request) {
            const user = db.users.find(u => u.username === request.user);
            if (user && user.balance >= request.amount) {
                user.balance -= request.amount;
                request.status = 'approved';
                saveDB();
                showMessage(`উইথড্র অনুমোদিত।`, 'success');
                updateDashboard();
            } else { showMessage('ইউজারের অপর্যাপ্ত ব্যালেন্স!', 'error'); }
        }
    }
    function rejectWithdrawal(reqId) {
        const request = db.withdrawalRequests.find(r => r.id === reqId);
        if (request) { request.status = 'rejected'; saveDB(); showMessage(`উইথড্র বাতিল।`, 'error'); updateDashboard(); }
    }

    function populateUserSelect() {
        const userSelect = document.getElementById('userSelect');
        userSelect.innerHTML = '<option value="">ইউজার নির্বাচন করুন</option>';
        db.users.filter(u => !u.isAdmin).forEach(user => {
            userSelect.innerHTML += `<option value="${user.username}">${user.username} (ব্যালেন্স: ${user.balance} BDT)</option>`;
        });
    }

    function addBalanceToUser() {
        const username = document.getElementById('userSelect').value;
        const amount = parseInt(document.getElementById('addBalanceAmount').value);
        if (!username || !amount || amount <= 0) { showMessage('ইউজার এবং সঠিক পরিমাণ নির্বাচন করুন।', 'error'); return; }
        const user = db.users.find(u => u.username === username);
        if (user) { user.balance += amount; saveDB(); showMessage(`${username}-এর অ্যাকাউন্টে ${amount} BDT যোগ হয়েছে।`, 'success'); updateDashboard(); }
    }

    function displayAllUsers() {
        const allUsersDiv = document.getElementById('allUsersList');
        const normalUsers = db.users.filter(u => !u.isAdmin);
        if (normalUsers.length === 0) { allUsersDiv.innerHTML = '<p>কোনো ইউজার নেই।</p>'; return; }
        let html = `<table class="data-table"><tr><th>ইউজারনেম</th><th>ব্যালেন্স</th><th>অ্যাকশন</th></tr>`;
        normalUsers.forEach(user => {
            html += `<tr><td>${user.username}</td><td>${user.balance} BDT</td><td class="actions">
                <button class="delete" onclick="deleteUser(${user.id})">ডিলিট</button></td></tr>`;
        });
        html += `</table>`;
        allUsersDiv.innerHTML = html;
    }

    function deleteUser(userId) {
        if (confirm('আপনি কি নিশ্চিত যে আপনি এই ইউজারকে ডিলিট করতে চান?')) {
            db.users = db.users.filter(u => u.id !== userId);
            saveDB();
            showMessage('ইউজার সফলভাবে ডিলিট করা হয়েছে।', 'success');
            updateDashboard();
        }
    }

    function displayAllItemsForAdmin() {
        const allItemsDiv = document.getElementById('allItemsList');
        if (db.items.length === 0) { allItemsDiv.innerHTML = '<p>কোনো অফার নেই।</p>'; return; }
        let html = `<table class="data-table"><tr><th>অফারের নাম</th><th>পুরস্কার</th><th>অ্যাকশন</th></tr>`;
        db.items.forEach(item => {
            html += `<tr><td>${item.name}</td><td>${item.price} BDT</td><td class="actions">
                <button class="delete" onclick="deleteItem(${item.id})">ডিলিট</button></td></tr>`;
        });
        html += `</table>`;
        allItemsDiv.innerHTML = html;
    }

    function addNewItem() {
        const name = document.getElementById('newItemName').value;
        const price = parseInt(document.getElementById('newItemPrice').value);
        if (!name || !price || price <= 0) { showMessage('অফারের নাম এবং সঠিক পুরস্কারের পরিমাণ দিন।', 'error'); return; }
        const newItem = { id: nextId++, name, price };
        db.items.push(newItem);
        saveDB();
        showMessage('নতুন অফার যোগ করা হয়েছে।', 'success');
        document.getElementById('newItemName').value = '';
        document.getElementById('newItemPrice').value = '';
        updateDashboard();
    }

    function deleteItem(itemId) {
        if (confirm('আপনি কি নিশ্চিত যে আপনি এই অফারটি ডিলিট করতে চান?')) {
            db.items = db.items.filter(i => i.id !== itemId);
            saveDB();
            showMessage('অফার সফলভাবে ডিলিট করা হয়েছে।', 'success');
            updateDashboard();
        }
    }
