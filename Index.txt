        function showSection(sectionId) {
            // Hide all sections
            document.querySelectorAll('.content-section').forEach(section => {
                section.classList.remove('active');
            });
            
            // Show selected section
            document.getElementById(sectionId).classList.add('active');
            currentSection = sectionId;
            
            // Update footer navigation
            document.querySelectorAll('.footer-nav-item').forEach(item => {
                item.classList.remove('active');
            });
            
            // Find the corresponding footer item and mark as active
            const footerItems = document.querySelectorAll('.footer-nav-item');
            if (sectionId === 'dashboardSection') footerItems[0].classList.add('active');
            else if (sectionId === 'adsSection') footerItems[1].classList.add('active');
            else if (sectionId === 'referralSection') footerItems[2].classList.add('active');
            else if (sectionId === 'withdrawSection') footerItems[3].classList.add('active');
            else if (sectionId === 'profileSection') footerItems[4].classList.add('active');
        }

                return;

        // Show notification
        function showNotification(type, title, message) {
            const notification = document.getElementById('notification');
            const icon = document.getElementById('notificationIcon');
            const titleElement = document.getElementById('notificationTitle');
            const messageElement = document.getElementById('notificationMessage');
            
            // Set notification type
            notification.className = 'notification';
            notification.classList.add(type);
            
            // Set icon based on type
            if (type === 'success') {
                icon.className = 'fas fa-check-circle';
            } else if (type === 'error') {
                icon.className = 'fas fa-exclamation-circle';
            } else if (type === 'info') {
                icon.className = 'fas fa-info-circle';
            }
            
            // Set content
            titleElement.textContent = title;
            messageElement.textContent = message;
            
            // Show notification
            notification.classList.add('show');
            
            // Hide after 3 seconds
            setTimeout(() => {
                notification.classList.remove('show');
            }, 3000);
        }
        
        // Initialize scratch card
        document.addEventListener('DOMContentLoaded', function() {
            const scratchArea = document.querySelector('.scratch-area');
            const scratchOverlay = document.querySelector('.scratch-overlay');
            let isScratching = false;
            
            scratchArea.addEventListener('mousedown', function(e) {
                isScratching = true;
                scratch(e);
            });
            
            scratchArea.addEventListener('mousemove', function(e) {
                if (isScratching) {
                    scratch(e);
                }
            });
            
            scratchArea.addEventListener('mouseup', function() {
                isScratching = false;
            });
            
            scratchArea.addEventListener('touchstart', function(e) {
                isScratching = true;
                const touch = e.touches[0];
                const mouseEvent = new MouseEvent('mousedown', {
                    clientX: touch.clientX,
                    clientY: touch.clientY
                });
                scratchArea.dispatchEvent(mouseEvent);
            });
            
            scratchArea.addEventListener('touchmove', function(e) {
                if (isScratching) {
                    const touch = e.touches[0];
                    const mouseEvent = new MouseEvent('mousemove', {
                        clientX: touch.clientX,
                        clientY: touch.clientY
                    });
                    scratchArea.dispatchEvent(mouseEvent);
                }
            });
            
            scratchArea.addEventListener('touchend', function() {
                isScratching = false;
            });
            
            function scratch(e) {
                const rect = scratchArea.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                
                const scratchCtx = document.createElement('canvas').getContext('2d');
                const canvas = document.createElement('canvas');
                canvas.width = scratchArea.offsetWidth;
                canvas.height = scratchArea.offsetHeight;
                canvas.style.position = 'absolute';
                canvas.style.top = '0';
                canvas.style.left = '0';
                canvas.style.pointerEvents = 'none';
                
                const ctx = canvas.getContext('2d');
                ctx.globalCompositeOperation = 'destination-out';
                ctx.beginPath();
                ctx.arc(x, y, 20, 0, Math.PI * 2);
                ctx.fill();
                
                if (!scratchArea.querySelector('canvas')) {
                    scratchArea.appendChild(canvas);
                } else {
                    const existingCanvas = scratchArea.querySelector('canvas');
                    const existingCtx = existingCanvas.getContext('2d');
                    existingCtx.globalCompositeOperation = 'destination-out';
                    existingCtx.beginPath();
                    existingCtx.arc(x, y, 20, 0, Math.PI * 2);
                    existingCtx.fill();
                }
                
                // Check if 50% is scratched
                const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
                const pixels = imageData.data;
                let transparent = 0;
                
                for (let i = 3; i < pixels.length; i += 4) {
                    if (pixels[i] < 128) {
                        transparent++;
                    }
                }
                
                if (transparent > pixels.length / 8) {
                    scratchOverlay.style.opacity = '0';
                    showNotification('success', 'অভিনন্দন', 'স্ক্র্যাচ করে ৳ 10 আয় করেছেন');
                }
            }
        });
    </script>
